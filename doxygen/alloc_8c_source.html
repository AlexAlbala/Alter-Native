28 /*
29  * Separate free lists are maintained for different sized objects
30  * up to MAXOBJBYTES.
31  * The call GC_allocobj(i,k) ensures that the freelist for
32  * kind k objects of size i points to a non-empty
33  * free list. It returns a pointer to the first entry on the free list.
34  * In a single-threaded world, GC_allocobj may be called to allocate
35  * an object of (small) size i as follows:
36  *
37  * opp = &(GC_objfreelist[i]);
38  * if (*opp == 0) GC_allocobj(i, NORMAL);
39  * ptr = *opp;
40  * *opp = obj_link(ptr);
41  *
42  * Note that this is very fast if the free list is non-empty; it should
43  * only involve the execution of 4 or 5 simple instructions.
44  * All composite objects on freelists are cleared, except for

204 # ifdef STACK_GROWS_UP
205  word stack_size = (ptr_t)(&dummy) - GC_stackbottom;
206 # else
207  word stack_size = GC_stackbottom - (ptr_t)(&dummy);
208 # endif

238 /* Return the number of bytes allocated, adjusted for explicit storage */
239 /* management, etc.. This number is used in deciding when to trigger */
240 /* collections. */
241 STATIC word GC_adj_bytes_allocd(void)
242 {
243  signed_word result;
244  signed_word expl_managed = (signed_word)GC_non_gc_bytes
245  - (signed_word)GC_non_gc_bytes_at_gc;
246 
247  /* Don't count what was explicitly freed, or newly allocated for */
248  /* explicit management. Note that deallocating an explicitly */
249  /* managed object should not alter result, assuming the client */
250  /* is playing by the rules. */
251  result = (signed_word)GC_bytes_allocd
252  + (signed_word)GC_bytes_dropped
253  - (signed_word)GC_bytes_freed
254  + (signed_word)GC_finalizer_bytes_freed
255  - expl_managed;
256  if (result > (signed_word)GC_bytes_allocd) {
257  result = GC_bytes_allocd;
258  /* probably client bug or unfortunate scheduling */
259  }
260  result += GC_bytes_finalized;
261  /* We count objects enqueued for finalization as though they */
262  /* had been reallocated this round. Finalization is user */
263  /* visible progress. And if we don't count this, we have */
264  /* stability problems for programs that finalize all objects. */
265  if (result < (signed_word)(GC_bytes_allocd >> 3)) {
266  /* Always count at least 1/8 of the allocations. We don't want */
267  /* to collect too infrequently, since that would inhibit */
268  /* coalescing of free storage blocks. */
269  /* This also makes us partially robust against client bugs. */
270  return(GC_bytes_allocd >> 3);
271  } else {
272  return(result);
273  }
274 }

415 # endif

501 {
502  int i;
503  IF_CANCEL(int cancel_state;)
504 
505  if (GC_dont_gc) return;
506  DISABLE_CANCEL(cancel_state);
507  if (GC_incremental && GC_collection_in_progress()) {
508  for (i = GC_deficit; i < GC_RATE*n; i++) {
509  if (GC_mark_some((ptr_t)0)) {
510  /* Need to finish a collection */
511 # ifdef SAVE_CALL_CHAIN
512  GC_save_callers(GC_last_stack);
513 # endif
514 # ifdef PARALLEL_MARK

516  GC_wait_for_reclaim();
517 # endif
518  if (GC_n_attempts < MAX_PRIOR_ATTEMPTS
519  && GC_time_limit != GC_TIME_UNLIMITED) {
520 # ifndef NO_CLOCK
521  GET_TIME(GC_start_time);
522 # endif
523  if (!GC_stopped_mark(GC_timeout_stop_func)) {
524  GC_n_attempts++;
525  break;
526  }
527  } else {
528  /* FIXME: If possible, GC_default_stop_func should be */
529  /* used here. */
530  (void)GC_stopped_mark(GC_never_stop_func);
531  }
532  GC_finish_collection();
533  break;
534  }
535  }
536  if (GC_deficit > 0) GC_deficit -= GC_RATE*n;
537  if (GC_deficit < 0) GC_deficit = 0;
538  } else {
539  GC_maybe_gc();
540  }
541  RESTORE_CANCEL(cancel_state);

1040  * Assumes p is HBLKSIZE aligned, and bytes is a multiple of HBLKSIZE.
1041  */
1042 GC_INNER void GC_add_to_heap(struct hblk *p, size_t bytes)
1043 {

1045  word endp;
1046 
1047  if (GC_n_heap_sects >= MAX_HEAP_SECTS) {
1048  ABORT("Too many heap sections: Increase MAXHINCR or MAX_HEAP_SECTS");
1049  }

1051  /* Can't handle memory near address zero. */
1052  ++p;
1053  bytes -= HBLKSIZE;
1054  if (0 == bytes) return;
1055  }
1056  endp = (word)p + bytes;
1057  if (endp <= (word)p) {
1058  /* Address wrapped. */
1059  bytes -= HBLKSIZE;
1060  if (0 == bytes) return;
1061  endp -= HBLKSIZE;

1064  if (0 == phdr) {

1067  /* which is entirely appropriate. */
1068  return;

1071  GC_heap_sects[GC_n_heap_sects].hs_start = (ptr_t)p;
1072  GC_heap_sects[GC_n_heap_sects].hs_bytes = bytes;
1073  GC_n_heap_sects++;

1075  phdr -> hb_flags = 0;

1077  GC_heapsize += bytes;

1082  /* us from getting a false hit from the variable */
1083  /* itself. There's some unintentional reflection */
1084  /* here. */
1085  }
1086  if ((ptr_t)p + bytes >= (ptr_t)GC_greatest_plausible_heap_addr) {
1087  GC_greatest_plausible_heap_addr = (void *)endp;
1088  }
1089 }

1133 /*
1134  * this explicitly increases the size of the heap. It is used
1135  * internally, but may also be invoked from GC_expand_hp by the user.
1136  * The argument is in units of HBLKSIZE.
1137  * Tiny values of n are rounded up.
1138  * Returns FALSE on failure.
1139  */
1140 GC_INNER GC_bool GC_expand_hp_inner(word n)
1141 {
1142  word bytes;
1143  struct hblk * space;
1144  word expansion_slop; /* Number of bytes by which we expect the */
1145  /* heap to expand soon. */
1146 
1147  if (n < MINHINCR) n = MINHINCR;
1148  bytes = n * HBLKSIZE;
1149  /* Make sure bytes is a multiple of GC_page_size */
1150  {
1151  word mask = GC_page_size - 1;
1152  bytes += mask;
1153  bytes &= ~mask;
1154  }
1155 
1156  if (GC_max_heapsize != 0 && GC_heapsize + bytes > GC_max_heapsize) {
1157  /* Exceeded self-imposed limit */
1158  return(FALSE);
1159  }
1160  space = GET_MEM(bytes);
1161  GC_add_to_our_memory((ptr_t)space, bytes);
1162  if (space == 0) {
1163  if (GC_print_stats) {
1164  GC_log_printf("Failed to expand heap by %ld bytes\n",
1165  (unsigned long)bytes);
1166  }
1167  return(FALSE);
1168  }
1169  if (GC_print_stats) {
1170  GC_log_printf("Increasing heap size by %lu after %lu allocated bytes\n",
1171  (unsigned long)bytes, (unsigned long)GC_bytes_allocd);
1172  }
1173  /* Adjust heap limits generously for blacklisting to work better. */
1174  /* GC_add_to_heap performs minimal adjustment needed for */
1175  /* correctness. */
1176  expansion_slop = min_bytes_allocd() + 4*MAXHINCR*HBLKSIZE;
1177  if ((GC_last_heap_addr == 0 && !((word)space & SIGNB))
1178  || (GC_last_heap_addr != 0 && GC_last_heap_addr < (ptr_t)space)) {
1179  /* Assume the heap is growing up */
1180  word new_limit = (word)space + bytes + expansion_slop;
1181  if (new_limit > (word)space) {
1182  GC_greatest_plausible_heap_addr =
1183  (void *)GC_max((word)GC_greatest_plausible_heap_addr,
1184  (word)new_limit);
1185  }
1186  } else {
1187  /* Heap is growing down */
1188  word new_limit = (word)space - expansion_slop;
1189  if (new_limit < (word)space) {
1190  GC_least_plausible_heap_addr =
1191  (void *)GC_min((word)GC_least_plausible_heap_addr,
1192  (word)space - expansion_slop);
1193  }
1194  }
1195  GC_prev_heap_addr = GC_last_heap_addr;
1196  GC_last_heap_addr = (ptr_t)space;
1197  GC_add_to_heap(space, bytes);
1198  /* Force GC before we are likely to allocate past expansion_slop */
1199  GC_collect_at_heapsize =
1200  GC_heapsize + expansion_slop - 2*MAXHINCR*HBLKSIZE;
1201  if (GC_collect_at_heapsize < GC_heapsize /* wrapped */)
1202  GC_collect_at_heapsize = (word)(-1);
1203  return(TRUE);
1204 }
