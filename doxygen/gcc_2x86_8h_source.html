134 AO_INLINE int
135 AO_compare_double_and_swap_double_full(volatile AO_double_t *addr,
136  AO_t old_val1, AO_t old_val2,
137  AO_t new_val1, AO_t new_val2)
138 {
139  char result;
140 #if __PIC__
141  /* If PIC is turned on, we can't use %ebx as it is reserved for the
142  GOT pointer. We can save and restore %ebx because GCC won't be
143  using it for anything else (such as any of the m operands) */
144  /* We use %edi (for new_val1) instead of a memory operand and swap */
145  /* instruction instead of push/pop because some GCC releases have */
146  /* a bug in processing memory operands (if address base is %esp) in */
147  /* the inline assembly after push. */
148  __asm__ __volatile__("xchg %%ebx,%6;" /* swap GOT ptr and new_val1 */
149  "lock; cmpxchg8b %0; setz %1;"
150  "xchg %%ebx,%6;" /* restore ebx and edi */
151  : "=m"(*addr), "=a"(result)
152  : "m"(*addr), "d" (old_val2), "a" (old_val1),
153  "c" (new_val2), "D" (new_val1) : "memory");
154 #else
155  /* We can't just do the same thing in non-PIC mode, because GCC
156  * might be using %ebx as the memory operand. We could have ifdef'd
157  * in a clobber, but there's no point doing the push/pop if we don't
158  * have to. */
159  __asm__ __volatile__("lock; cmpxchg8b %0; setz %1;"
160  : "=m"(*addr), "=a"(result)
161  : "m"(*addr), "d" (old_val2), "a" (old_val1),
162  "c" (new_val2), "b" (new_val1) : "memory");
163 #endif
164  return (int) result;
165 }
